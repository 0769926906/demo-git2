git clone: kéo dự án trên sever về máy tính 
git init: tạo thư mục git local
git status: để kiểm tra status(vùng) của file
git add . : đưa tất cả file có sự thay đổi vào vùng staging
git add [ten_file]: đưa một file có sự thay đổi vào vùng staging
git rm --cached <file> : đưa một file từ staging quay ngược lại vùng working directory
git config --global user.email "longta10a2@gmail.com": set email(thay đổi email phải khai báo lại)
git config --global user.name "Long" : set tên
git commit -m "lời nhắn": đưa file có sự thay đổi sang vùng git kèm 1 lời nhắn (local repo)
git remote add origin "link": tạo liên kết giữa máy và sever
git push: đẩy code từ local lên sever
git branch <tên branch mới>: tạo nhánh mới
git checkout <branch name>: chuyển từ branch master sang branch mới
git checkout -b <tên branch mới>: tạo branch mới và chuyển qua branch mới luôn
